### Promise

之前都是基于单个Promise实例运行，现考虑多个Promise实例运行情况。

```
promise.then(f1).then(f2).then(f3);
1. 如果f1也是异步任务，那么注册的f2应在f1的promise的onFulfilled中，以此类推。
2. 既要考虑同步，也要考虑异步，那么让then返回一个新的Promise即可。
```

**支持串行任务**

- 支持串行任务，每次then返回一个新的Promise（值为注册then的unfulfilled函数的返回值）


```js
// 支持同步和异步任务
var PENDING = 'pending';
var FULFILLED = 'fulfilled';
var REJECTED = 'rejected';

function MyPromise(excutor) {
  var self = this;
  self.value = null;
  self.reason = null;
  self.status = PENDING;
  self.onFulfilled = [];
  self.onRejected = [];
  function resolve(value) {
    setTimeout(function() {
      if (self.status === PENDING) { //事件循环机制，解决同时调用resolve和reject可以触发两次状态更改
        self.value = value;
        self.status = FULFILLED;
        for (var i = 0; i < self.onFulfilled.length; i++) {
          self.onFulfilled[i](self.value);
        }
      }
    }); 
  }
  function reject(reason) {
    setTimeout(function() {
      if (self.status === PENDING) {
        self.reason = reason;
        self.status = REJECTED;
        for (var i = 0; i < self.onRejected.length; i++) {
          self.onRejected[i](self.reason);
        }
      }
    });
  }
  try {
    excutor(resolve, reject);
  } catch (e) {
    reject(e);
  }
}
MyPromise.prototype.then = function(onFulfilled, onRejected) {
  var self = this;
  if (this.status === PENDING) {
    return new MyPromise(function(resolve, reject) {
      self.onFulfilled.push(function (value) {
        try {
          var x = onFulfilled(value);
          resolvePromise(x, resolve, reject);
        } catch (e) {
          reject(e);
        }
      });
      self.onRejected.push(function (reason) {
        try {
          var x = onRejected(reason);
          resolvePromise(x, resolve, reject);
        } catch (e) {
          reject(e);
        }
      });
    });
  }
}

function resolvePromise(x, resolve, reject) {
  if (x instanceof MyPromise) {
    x.then(function (y) { // y仍有可能是PROMISE
      resolvePromise(y, resolve, reject);
    })
  } else {
    resolve(x);
  }
}

// TEST
new MyPromise(function (resolve) {
  resolve(1)
})
  .then(function (value) {
  return new MyPromise(function (resolve) {
    new MyPromise(function (resolve) {
      resolve(2 + value);
    }).then(resolve);
  })
})
  .then(function (a) {
  console.log(a);
})


new MyPromise(function (resolve) {
  resolve(1)
})
  .then(function (value) {
  return new MyPromise(function (resolve) {
    setTimeout(function () {
      new MyPromise(function (resolve) {
        setTimeout(function () {
          resolve(2 + value);
        }, 1000)
      }).then(resolve);
    }, 1000)
  })
})
  .then(function (a) {
  console.log(a);
})

```
